FusionScript is a scripting language inspired by Python, C++, and Lua and runs on a modified Lua runtime.

Code Example

 1 class Account {
 2     __new(balance = 0)=> {
 3         @balance = balance;
 4     }
 5 
 6     deposit(amount)=> {
 7         @balance = (+ @balance amount);
 8         return true;
 9     }
10 
11     withdraw(amount)=> {
12         if (> amount @balance)
13             return false;
14         else {
15             @balance = (- @balance amount);
16             return true;
17         }
18     }
19 
20     balance()=>
21         return @balance;
22 }
23 
24 bob = Account(500); -- 500
25 bob:deposit(600);   -- 1100
26 bob:withdraw(1000); -- 100
27 assert(bob:withdraw(math.max)); -- errors
